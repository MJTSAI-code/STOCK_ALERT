import json import time import os from datetime import datetime import pytz import yfinance as yf STOCKS = ['NOW', 'NVDA', 'LITE', 'ONDS', 'MRVL', 'GOOG'] def is_news_catching_window(): """⏰ 早上 04:30-05:30 / 晚上 16:30-17:30 才會去 Yahoo 抓新新聞""" tz = pytz.timezone('Asia/Taipei') now_local = datetime.now(tz) current_time_str = now_local.strftime("%H:%M") is_morning_window = ("04:30" <= current_time_str <= "05:30") is_evening_window = ("16:30" <= current_time_str <= "17:30") if is_morning_window or is_evening_window: print(f"⏰ 台北時間 {current_time_str} 【黃金一小時】更新即時新聞") return True else: print(f"🔒 台北時間 {current_time_str} 非準備時段，鎖定並沿用歷史新聞") return False def load_existing_news(): """💾 讀取現有的 json，完整繼承上一次抓到的真新聞""" existing_news_map = {symbol: [] for symbol in STOCKS} filename = 'stock_data.json' if os.path.exists(filename): try: with open(filename, 'r', encoding='utf-8') as f: old_data = json.load(f) data_block = old_data.get('data', {}) for symbol in STOCKS: if symbol in data_block and 'news' in data_block[symbol]: existing_news_map[symbol] = data_block[symbol]['news'] except Exception as e: print(f"⚠️ 讀取舊 JSON 失敗: {e}") return existing_news_map def fetch_stock_data(): output_data = {} allow_news_fetch = is_news_catching_window() history_news_map = load_existing_news() for symbol in STOCKS: try: ticker = yf.Ticker(symbol) # 1. 價格部分：每次執行都拿最新報價 info = ticker.info current_price = info.get('regularMarketPrice') or info.get('currentPrice') or 0.0 dollar_change = info.get('regularMarketChange') or 0.0 percent_change = info.get('regularMarketChangePercent') or 0.0 if current_price == 0.0: hist = ticker.history(period="1d") if not hist.empty: current_price = float(hist['Close'].iloc[-1]) open_p = float(hist['Open'].iloc[-1]) dollar_change = current_price - open_p percent_change = (dollar_change / open_p) * 100 if open_p != 0 else 0.0 # 2. 新聞部分：實施時段防守策略 processed_news = [] if allow_news_fetch: raw_news = ticker.news if raw_news: for item in raw_news[:3]: # 🎯 改用最新結構：相容新版 yfinance content = item.get('content', {}) headline = content.get('title') or item.get('title') or "" news_time = content.get('pubDate') or item.get('providerPublishTime') # 🎯 核心修復：抓取 clickThroughUrl 的真正真新聞網址 click_through = content.get('clickThroughUrl', {}) if content else {} news_url = click_through.get('url') or item.get('link') or item.get('url') or "" if headline and news_url and news_url != "#" and news_time: headline_str = headline.strip() if "無新聞標題" not in headline_str and "無最新新聞" not in headline_str: # 時間戳記轉換 if isinstance(news_time, str): try: dt = datetime.fromisoformat(news_time.replace('Z', '+00:00')) t_stamp = int(dt.timestamp()) except: t_stamp = int(time.time()) else: t_stamp = int(news_time) processed_news.append({ "headline": headline_str, "datetime": t_stamp, "source": content.get('provider', {}).get('displayName') or item.get('publisher') or "Yahoo Finance", "url": news_url.strip() # 直達真正的第三方新聞網站 }) if not processed_news: processed_news = history_news_map.get(symbol, []) else: # 🌟 非黃金時段：百分之百原封不動，沿用上一次抓到的歷史真新聞 processed_news = history_news_map.get(symbol, []) output_data[symbol] = { "quote": { "c": float(current_price), "d": float(dollar_change), "dp": float(percent_change) }, "news": processed_news } time.sleep(1.5) except Exception as e: print(f"❌ 處理 {symbol} 出現異常: {str(e)}") output_data[symbol] = { "quote": {"c": 0.0, "d": 0.0, "dp": 0.0}, "news": history_news_map.get(symbol, []) } final_json = { "name": "美股盤前情報資料庫", "updated_at": datetime.utcnow().isoformat() + "+00:00", "data": output_data } with open('stock_data.json', 'w', encoding='utf-8') as f: json.dump(final_json, f, ensure_ascii=False, indent=2) if __name__ == "__main__": fetch_stock_data()
